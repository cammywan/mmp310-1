Okay, let's try a couple more lines:

document.body
document.body.style
document.body.style.color = "purple";

We could write our entire HTML document this way...

var elem = document.createElement("h1");
elem.innerHTML = "JavaScript is kewl!";
document.body.appendChild(elem);

...but its easier to write some HTML and then modify it with JavaScript.

For every node in the DOM, there are many properties and methods we can use to create, remove and modify elements of our HTML document.

Referencing elements in the DOM:

document.getElementById("container");
document.getElementsByClassName("slide");
document.getElementsByTagName("p");
document.querySelector(".slide");
document.querySelector("#container");
document.querySelector(".banana");

Note the spelling and punctuation of these methods, they have to be written exactly right.

.getElement is for ids, vs .getElements for class and tag names

Creating and appending elements to DOM:

var newslide = document.createElement("div");
newslide.className = "slide";
newslide.id = "newslide";
document.body.appendChild(newslide);

And removing them:

var newslide = document.getElementById("newslide");
document.body.removeChild(newslide);

We can access child and parent nodes:

var c = document.querySelector("#container");
console.log(c.firstChild);
console.log(c.lastChild);
console.log(c.childNodes);

And get the parent node or sibbling node:

var children = c.childNodes;
console.log(children[0].parentNode);
console.log(children[0].previousSibling);
console.log(children[0].nextSibling);

Creating and modifying style and content:

var newslide = document.createElement("div");
newslide.className = "slide";
newslide.innerHTML = "This is a new slide!";
newslide.style.color = "red";
c.appendChild(newslide);

You can access any style in any element:

var c = document.getElementById("container");
c.style.color = "green";

Add an image to the document:

var img = new Image();
img.src = "https://i.ytimg.com/vi/tntOCGkgt98/maxresdefault.jpg";
var elem = document.createElement("div");
elem.appendChild(img);
c.appendChild(elem);
